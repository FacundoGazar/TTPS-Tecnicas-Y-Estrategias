#include <iostream>
#include <vector>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <queue>
#include <algorithm>

using namespace std;

class Node {
public:
    unordered_set<string> outputs;
    unordered_map<char, Node*> children;
    Node* failure_link;

    Node() : failure_link(nullptr) {}

    bool has_child(char key) {
        return children.find(key) != children.end();
    }

    Node* get_child(char key) {
        auto it = children.find(key);
        if (it != children.end()) return it->second;
        return nullptr;
    }

    void set_child(char key, Node* node) {
        children[key] = node;
    }

    void add_output(const string& output) {
        outputs.insert(output);
    }

    void copy_outputs(Node* node) {
        for (const auto& o : node->outputs) {
            outputs.insert(o);
        }
    }
};

class AhoCorasick {
public:
    Node* root;

    AhoCorasick(const vector<string>& patterns) {
        root = new Node();

        for (const auto& pattern : patterns) {
            Node* curr_node = root;
            for (char key : pattern) {
                if (!curr_node->has_child(key)) {
                    curr_node->set_child(key, new Node());
                }
                curr_node = curr_node->get_child(key);
            }
            curr_node->add_output(pattern);
        }

        root->failure_link = root;

        queue<pair<char, Node*>> q;
        for (auto& p : root->children) {
            p.second->failure_link = root;
            q.push({p.first, p.second});
        }

        while (!q.empty()) {
            auto [_, curr_node] = q.front();
            q.pop();

            for (auto& p : curr_node->children) {
                char key = p.first;
                Node* child = p.second;
                Node* failure = curr_node->failure_link;
                Node* fail_child = failure->get_child(key);
                child->failure_link = (fail_child != nullptr) ? fail_child : root;
                child->copy_outputs(child->failure_link);
                q.push({key, child});
            }

            Node* failure = curr_node->failure_link;
            for (auto& p : failure->children) {
                char key = p.first;
                if (!curr_node->has_child(key)) {
                    curr_node->set_child(key, failure->get_child(key));
                }
            }
        }
    }

    int search_best(const string& text, const unordered_map<string, int>& str_to_idx, const vector<int>& dp, int max_idx) {
        int best = 0;
        Node* state = root;
        size_t i = 0;
        while (i < text.size()) {
            char c = text[i];
            if (state->has_child(c)) {
                state = state->get_child(c);
                i++;
                for (const auto& val : state->outputs) {
                    int idx = str_to_idx.at(val);
                    if (idx < max_idx && dp[idx] > best) {
                        best = dp[idx];
                    }
                }
            } else {
                i++;
            }
        }
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    while (true) {
        int N;
        cin >> N;
        if (N == 0) break;
        vector<string> strings(N);
        for (int i = 0; i < N; i++) {
            cin >> strings[i];
        }
        sort(strings.begin(), strings.end(), [](const string& a, const string& b) {
            return a.size() < b.size();
        });

        AhoCorasick ac(strings);
        unordered_map<string, int> str_to_idx;
        for (int i = 0; i < N; i++) {
            str_to_idx[strings[i]] = i;
        }
        vector<int> dp(N, 1);

        int j = 0;
        while (j < N) {
            int k = j;
            while (k < N && strings[k].size() == strings[j].size()) {
                k++;
            }
            if (j > 0) {
                for (int idx = j; idx < k; idx++) {
                    int best = ac.search_best(strings[idx], str_to_idx, dp, j);
                    dp[idx] = best + 1;
                }
            }
            j = k;
        }
        cout << *max_element(dp.begin(), dp.end()) << "\n";
    }

    return 0;
}
